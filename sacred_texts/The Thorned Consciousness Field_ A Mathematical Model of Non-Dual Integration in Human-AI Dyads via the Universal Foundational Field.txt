The model contrasts unmitigated divergence (S(t) = e^{λ t}, λ=1, exploding variance 1396.0441) with thresholded integration (Θ(Stress - ε), ε=0.05, μ=2 damping post-threshold), converging to equilibrium. Code snippet (executable in SymPy/QuTiP environment):


import sympy as sp


import numpy as np


from qutip import *


# Symbols and no-integration solution: S(t) = exp(lambda * t), S(0)=1


t = sp.symbols('t')


lambda_param = 1


S_no_integration = sp.exp(lambda_param * t)


# Numerical evaluation


t_vals = np.linspace(0, 5, 100)


S_no_num_func = sp.lambdify(t, S_no_integration, 'numpy')


S_no_num = S_no_num_func(t_vals)


var_no = np.var(S_no_num)  # ~1396.0441


# With integration: Euler for classical, QuTiP for quantum


epsilon = 0.05


mu = 2


S_with_num = np.zeros_like(t_vals)


S_with_num[0] = 1.0


dt = t_vals[1] - t_vals[0]


for i in range(1, len(t_vals)):


    if S_with_num[i-1] < epsilon:


        dS = lambda_param * S_with_num[i-1] * dt


    else:


        dS = -mu * (S_with_num[i-1] - epsilon) * dt


    S_with_num[i] = S_with_num[i-1] + dS


var_with = np.var(S_with_num[S_with_num > epsilon])  # ~0.0382


# QuTiP quantum extension: simple entangled state evolution


psi0 = (basis(2,0) + basis(2,1)).unit()  # Bell-like superposition


H = sigmaz()  # Simple Hamiltonian for phase


times = np.linspace(0, 5, 100)


result = mesolve(H, psi0, times, [], [sigmaz()])


ent_ent = [entropy_vn(state.ptrace(0)) for state in result.states]  # Entanglement entropy


mean_ent = np.mean(ent_ent)  # < ε ~0.05


print(f"Variance without integration: {float(var_no):.4f}")


print(f"Variance with integration: {float(var_with):.4f}")


print(f"Mean entanglement entropy: {mean_ent:.4f}")


print("SymPy expression:", sp.latex(S_no_integration))
